event DamagedPlayer
{
	Animate DamageAccountValue		Alpha	"255"	linear 0.0 0.0
	Animate DamageAccountValue		Alpha	"0"		linear 10.0 0.2
	Animate DamageAccountValueBG	Alpha	"255"	linear 0.0 0.0
	Animate DamageAccountValueBG	Alpha	"0"		linear 10.0 0.2
	
	Animate CrosshairXX	FgColor	"R G B Opacity"	linear 0.0 0.00001
	Animate CrosshairXX	FgColor	"R G B Opacity"	linear 0.15 0.15001
	// The first line animates your HUD crosshair when you hit someone.
	// The second line changes your HUD crosshair's colour back.
	// Replace "CrosshairXX" with your HUD crosshair's fieldname.
	// Edit the colours (R G B Opacity) to your liking.
}

event HudHealthBonusPulse
{
	Animate PlayerStatusHealthBonusImage	Alpha	"255"	linear 0.0 0.2
	Animate PlayerStatusHealthBonusImage	Alpha	"0"		linear 0.2 0.4
	
	Animate PlayerStatusHealthvalue			FgColor	"buffed HP"				linear 0.0 0.0
	Animate PlayerStatusHealthvalueShadow	FgColor	"buff shadow"			linear 0.0 0.0
	Animate PlayerStatusHealthvalueTargetID	FgColor	"TargetID buffed HP"	linear 0.0 0.0
	
	RunEvent HudHealthBonusPulseLoop 0.4
}
